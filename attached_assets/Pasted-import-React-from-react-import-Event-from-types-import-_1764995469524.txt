import React from 'react';
import { Event } from '../types';
import GlassCard from './ui/GlassCard';
import { formatCurrency } from '../services/mockDatabase';
import { CheckBadgeIcon } from './Icons';

interface EventCardProps {
  event: Event;
  onSelect: (event: Event) => void;
}

const EventCard: React.FC<EventCardProps> = ({ event, onSelect }) => {
  
  const handleClick = (e?: React.MouseEvent) => {
    e?.stopPropagation();
    onSelect(event);
  };

  // Determine lowest price to display "À partir de"
  const displayPrice = event.ticketTiers && event.ticketTiers.length > 0 
    ? Math.min(...event.ticketTiers.map(t => t.price))
    : event.price;

  return (
    <GlassCard 
      intensity="low"
      onClick={() => handleClick()}
      className="group rounded-[32px] cursor-pointer h-96 flex flex-col justify-end p-5 hover:bg-white/15 overflow-hidden transition-all duration-300 hover:shadow-[0_0_30px_rgba(255,255,255,0.1)] relative"
    >
      {/* Background Image with Overlay */}
      <div className="absolute inset-0 z-0">
        <img 
          src={event.imageUrl} 
          alt={event.title} 
          className="w-full h-full object-cover transition-transform duration-700 ease-out group-hover:scale-110"
        />
        <div className="absolute inset-0 bg-gradient-to-t from-black via-black/40 to-transparent opacity-90"></div>
      </div>

      {/* Verified Badge - Top Right Corner */}
      {event.isVerified && (
        <div className="absolute top-4 right-4 z-20 bg-white/10 backdrop-blur-md rounded-full p-1.5 border border-white/20 shadow-lg animate-pulse-slow">
           <CheckBadgeIcon className="w-5 h-5 text-blue-400" />
        </div>
      )}

      {/* Content */}
      <div className="relative z-10 transform transition-transform duration-500 ease-out group-hover:-translate-y-2">
        <div className="flex justify-between items-start mb-2">
          <span className="px-3 py-1 rounded-full bg-white/20 backdrop-blur-md text-xs font-semibold text-white border border-white/10 shadow-sm">
            {event.category}
          </span>
          <span className="text-orange-400 font-bold text-lg drop-shadow-md bg-black/50 px-2 py-1 rounded-lg backdrop-blur-sm">
            {formatCurrency(displayPrice)}
          </span>
        </div>
        
        <h3 className="text-2xl font-bold text-white mb-1 leading-tight group-hover:text-orange-400 transition-colors">
          {event.title}
        </h3>
        
        <p className="text-gray-300 text-sm mb-4 line-clamp-2">
          {new Date(event.date).toLocaleDateString('fr-FR', { weekday: 'long', day: 'numeric', month: 'long' })}
          <br/>
          <span className="text-gray-400">{event.location}</span>
        </p>

        <button 
          onClick={handleClick}
          className="w-full py-3 rounded-2xl bg-white text-black font-bold text-sm hover:bg-orange-500 hover:text-white transition-all duration-300 transform translate-y-4 opacity-0 group-hover:translate-y-0 group-hover:opacity-100 shadow-xl"
        >
          Réserver sa place
        </button>
      </div>
    </GlassCard>
  );
};

export default EventCard;
